```python
def binary_search(nums, target):
    left, right = 0, len(nums) - 1 # 循环不变量 双闭区间
    while left <= right: # 边界也要比 
        mid = (left + right) // 2
        # 必须列出所有情况
        if nums[mid] < target: 
            left = mid + 1  
        elif nums[mid] > target:
            right = mid - 1 
        elif nums[mid] == target:
            return mid
    return -1 
```

```python
def search(nums, target) -> int:
    # ------左边界------
    left,right=0,len(nums)-1 
    final_left=-1 
    while left<=right:
        mid=(left+right)//2
        if nums[mid]>target:
            right=mid-1
        elif nums[mid]<target:
            left=mid+1
        elif nums[mid]==target:
            final_left=mid  # 更新结果

    final_right=-1

        mid=(left+right)//2
        if nums[mid]>target:
            right=mid-1
        elif nums[mid]<target:
            left=mid+1
        elif nums[mid]==target:
            final_right=mid
            left=mid+1  # 寻找右边界，收缩左边界（唯一区别）
```

### 3. 排序算法

#### 快速排序QuickSort

时间复杂度O(nlogn) 

1. 确定基准值，左侧找小于基准值，右侧找大于基准值，持续交换
2. 将基准值 左右区间，递归排序。

```python
def swap(nums, left, right):
    """
    分区 选择一个基准元素,将数组分为两部分
    """
   # 以 nums[left] 作为基准数
   i,j=left,right
   base = nums[left]
   while i<j: # 等号无需交换
      while i < j and nums[i] <= base: # 左指针找到第一个大于基准的元素
         i += 1
      while i < j and base <= nums[j]: # 右指针找到第一个小于基准的元素 
         j -= 1
      if i < j: # 下标每次变动，都要判断
      	 nums[i], nums[j] = nums[j], nums[i] 
    nums[left], nums[i] = nums[i], nums[left] # 基准元素交换到对应位置，返回基准元素所在下标
    return i
def quicksort(nums, left, right):
    # 终止条件  等号无需交换
    if left >= right:
      	return 
    # 单层递归逻辑   先分区，再排序左右区间
    base_index = swap(nums, left, right) # 返回基准值下标 
    quicksort(nums, left, base_index-1)  
    quicksort(nums, base_index+1, right) 
```

```python
def levelOrder(root):
    # 边界条件   空树
    if root is None:
        return []
    dq = collections.deque([root]) # 1. 初始化队列，装入根节点
    result = [] # 保存结果
    while dq: 
        level = [] 
        for _ in range(len(dq)):   # 2. 遍历当前层，记录并装入左右孩子
            cur_node = dq.popleft() 
            level.append(cur_node.val)
            if cur_node.left: 
                dq.append(cur_node.left)
            if cur_node.right:
                dq.append(cur_node.right)
        result.append(level) # 3. 保存当前层结果
    return result
```

#### （1）跳跃游戏

题目：数组中的每个元素代表你在该位置可以跳跃的最大长度。判断是否能跳到最后一个下标。

```python
def jump(nums):
    cur_index=0
    for i in range(len(nums)):
        # 1. 终止条件 1）已越界  （2）能到达
        if i>cur_index:
            return False
        if cur_index>=len(nums)-1:
            return True
        # 更新当前最远下标
        cur_index=max(cur_index,i+nums[i])
    return False
```

#### （2） 跳跃游戏II：返回最小跳跃次数

1. 终止条件 （1）已越界 （2）能到达
2. 更新下步最远下标
3. 更新步数

```python
def jump(nums):
    if len(nums)<=1:
        return 0
    # 跳跃参数
    cur_index=0 # 当前最远下标
    next_index=0 # 增加一次步数的最远下标
    # 结果参数
    step=0

    for i in range(len(nums)):
        # 1. 终止条件 （1）next_index已越界   （2）cur_index能到达
        if i>next_index:
            return -1
        if cur_index>=len(nums)-1:
            return step

        # 2. 更新下步最远下标
        next_index=max(next_index,i+nums[i])
        # 3. 是否增加步数
        if i==cur_index:
            cur_index=next_index
            step+=1
    return step
```

# 数学

## 条件概率公式

$$
P(A \mid B) = \frac{P(A \cap B)}{P(B)}
$$

- $P(A|B)$      已知B发生的前提下，A发生的概率
- $P(A∩B)$   AB同时发生的概率
- $P(B)$           B发生的概率

## 独立性
事件A和B独立的定义是：一个事件的发生不影响另一个事件的发生。故$P(A \cap B) = P(A) \cdot P(B)$
